function getCurrentDateTime() {
    var now = new Date();

    var localDate = now.toLocaleString('en-US', { day: 'numeric', month: 'long', year: 'numeric' });
    var localTime = now.toLocaleString('en-US', { hour: 'numeric', minute: 'numeric', hour12: true });

    var rawTimezoneOffset = now.getTimezoneOffset();

    var timezoneOffsetHours = Math.floor(Math.abs(rawTimezoneOffset) / 60);
    var timezoneOffsetMinutes = Math.abs(rawTimezoneOffset) % 60;

    var timezoneOffsetString = (rawTimezoneOffset >= 0 ? '-' : '+') +
        ('00' + timezoneOffsetHours).slice(-2) +
        ':' +
        ('00' + timezoneOffsetMinutes).slice(-2);

    return localDate + ' at ' + localTime + ' (GMT ' + timezoneOffsetString + ')';
}

function updateLastUpdateTime() {
    var lastUpdateElement = document.getElementById('lastUpdate');
    if (lastUpdateElement) {
        lastUpdateElement.textContent = getCurrentDateTime();
    }
}

window.onload = function() {
    updateLastUpdateTime();
};

window.onbeforeunload = function() {
    updateLastUpdateTime();
